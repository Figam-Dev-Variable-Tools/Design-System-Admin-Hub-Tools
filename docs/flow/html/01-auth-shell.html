flowchart TD
    Start(["사용자 진입"]) --> Route{"요청 경로"}

    Route -->|"/login"| Login["로그인 화면<br/>셸 바깥 단독"]
    Route -->|"정의되지 않은 경로"| Fallback["/dashboard 로 리다이렉트"]
    Route -->|"그 외 전체"| Auth{"RequireAuth<br/>세션 있나"}

    Login --> Submit["아이디·비밀번호 제출"]
    Submit --> AuthOk{"인증 성공"}
    AuthOk -->|"실패"| LoginErr["오류 표시 · 폼 유지"]
    LoginErr --> Submit
    AuthOk -->|"성공"| ReturnUrl{"returnUrl 있나"}
    ReturnUrl -->|"있음"| BackTo["원래 가려던 경로로 복귀"]
    ReturnUrl -->|"없음"| Dash

    Auth -->|"세션 없음"| Redirect["/login?returnUrl=현재경로"]
    Redirect --> Login
    Auth -->|"세션 있음"| Shell["AppShell<br/>사이드바 + 헤더"]

    Shell --> Perm{"RequirePermission<br/>Outlet 을 통째로 감싼다"}
    Perm -->|"권한 없음"| Deny["403 화면"]
    Perm -->|"권한 있음"| Lazy["lazy 청크 로드<br/>Suspense 경계는 셸 안쪽"]

    Lazy --> Impl{"화면 구현됨"}
    Impl -->|"미구현 사이드바 리프"| Placeholder["준비 중 화면"]
    Impl -->|"구현됨"| Screen["요청한 화면"]

    Fallback --> Dash
    BackTo --> Screen
    Dash["대시보드<br/>/dashboard"] --> Screen

    Screen --> Err{"렌더 중 예외"}
    Err -->|"예외 발생"| Boundary["ErrorBoundary<br/>셸은 살리고 본문만 복구 UI"]
    Err -->|"정상"| Done(["화면 사용"])
    Boundary --> Retry["다시 시도"] --> Screen
